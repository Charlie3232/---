function doGet(e) { var action = e.parameter.action; if (action === 'getData') { return ContentService.createTextOutput(JSON.stringify(getAllData())) .setMimeType(ContentService.MimeType.JSON); } } function parseDateToYMD(raw) { if (raw == null || raw === "") return ""; if (raw instanceof Date) return Utilities.formatDate(raw, Session.getScriptTimeZone(), "yyyy-MM-dd"); let s = raw.toString().replace(/^'/, "").trim(); if (s.includes("T")) return s.split("T")[0]; if (s.includes("/")) return s.replace(/\//g, "-"); return s; } function parseTime(raw) { if (raw == null || raw === "") return ""; if (raw instanceof Date) return Utilities.formatDate(raw, Session.getScriptTimeZone(), "HH:mm"); return raw.toString().replace(/^'/, "").trim(); } function getAllData() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var issueSheet = ss.getSheetByName("Issues"); var macroSheet = ss.getSheetByName("巨集"); var calSheet = ss.getSheetByName("Calendar") || ss.insertSheet("Calendar"); var issueData = issueSheet.getDataRange().getValues(); var issues = []; for (var i = 1; i < issueData.length; i++) { issues.push({ issue: issueData[i][0], owner: issueData[i][1], status: issueData[i][2], customer: issueData[i][3], project: issueData[i][4], date: parseDateToYMD(issueData[i][5]), deadline: parseDateToYMD(issueData[i][6]), priority: issueData[i][7], description: issueData[i][8], records: issueData[i][9], link: issueData[i][10], id: issueData[i][11], actualClose: issueData[i][12] ? parseDateToYMD(issueData[i][12]) : "" // ✨ 新增讀取實際結案日 }); } var macroData = macroSheet.getDataRange().getValues(); var config = { owners: [], statusList: [], customers: [], projects: [], eventCategories: [] }; for (var j = 1; j < macroData.length; j++) { if (macroData[j][0]) config.owners.push(macroData[j][0]); if (macroData[j][1]) config.statusList.push(macroData[j][1]); if (macroData[j][2]) config.customers.push(macroData[j][2]); if (macroData[j][3]) config.projects.push(macroData[j][3]); if (macroData[j][5]) { config.eventCategories.push({ name: macroData[j][5].toString().trim(), color: macroData[j][6] ? macroData[j][6].toString().trim() : '灰色' }); } } var calData = calSheet.getDataRange().getValues(); var events = []; for (var k = 0; k < calData.length; k++) { if(calData[k][0] && calData[k][0].toString().startsWith("EV-")) { events.push({ id: calData[k][0].toString().replace(/^'/, ""), name: calData[k][1] ? calData[k][1].toString().replace(/^'/, "") : "", date: parseDateToYMD(calData[k][2]), time: parseTime(calData[k][3]), participants: calData[k][4] ? calData[k][4].toString() : "", location: calData[k][5] ? calData[k][5].toString() : "", endDate: parseDateToYMD(calData[k][6]), endTime: parseTime(calData[k][7]), category: calData[k][8] ? calData[k][8].toString().replace(/^'/, "") : "" }); } } return { issues: issues, config: config, events: events }; } function doPost(e) { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); if (data.action === "addEvent" || data.action === "editEvent") { var calSheet = ss.getSheetByName("Calendar"); var rowArray = [ data.id, data.name, "'" + data.date, "'" + data.time, data.participants, data.location, data.endDate ? "'" + data.endDate : "", data.endTime ? "'" + data.endTime : "", data.category ]; if (data.action === "addEvent") { calSheet.appendRow(rowArray); } else { var cDB = calSheet.getDataRange().getValues(); for (var c = 0; c < cDB.length; c++) { if (cDB[c][0] == data.id) { calSheet.getRange(c + 1, 1, 1, 9).setValues([rowArray]); break; } } } return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT); } else if (data.action === "deleteEvent") { var calSheet = ss.getSheetByName("Calendar"); var cDB2 = calSheet.getDataRange().getValues(); for (var c2 = 0; c2 < cDB2.length; c2++) { if (cDB2[c2][0] == data.id) { calSheet.deleteRow(c2 + 1); break; } } return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT); } var sheet = ss.getSheetByName("Issues"); if (data.action === "delete") { var db = sheet.getDataRange().getValues(); for (var i = 1; i < db.length; i++) { if (db[i][11] == data.id) { sheet.deleteRow(i + 1); break; } } return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT); } // ✨ 新增第 13 欄：實際結案日 var rowData = [ data.issue, data.owner, data.status, data.customer, data.project, "'" + data.date, "'" + data.deadline, data.priority, data.description, data.records, data.link, data.id, data.actualClose ? "'" + data.actualClose : "" ]; if (data.action === "add") { sheet.appendRow(rowData); } else if (data.action === "edit") { var db2 = sheet.getDataRange().getValues(); for (var j = 1; j < db2.length; j++) { if (db2[j][11] == data.id) { sheet.getRange(j + 1, 1, 1, 13).setValues([rowData]); break; } } } return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT); }
